Actor PPE_SSTommyGun : PPE_SeriousSamWeaponBase
{
	Tag "$TAG_SS_TG"
	Inventory.PickupSound "items/ssweapon"
	Weapon.SelectionOrder 800
	Weapon.SlotNumber 4
	Weapon.AmmoType "PPE_Bullets"
	Weapon.AmmoUse 1
	Weapon.AmmoGive 20
	Decal "BulletChip"
	DamageType "SeriousBullet"
	States
	{
	Drop:
	Spawn:
		SSTW A -1
		Stop
	Ready:
		SSTG A 1 A_WeaponReady
		Loop
	Deselect:
		SSTG A 1 A_Lower
		TNT1 A 0 A_Lower
		Loop
	Select:
		SSTG A 1 A_Raise
		TNT1 A 0 A_Raise
		Loop
	Fire:
		TNT1 A 0 A_GunFlash
		TNT1 A 0 A_AlertMonsters
		TNT1 A 0 A_PlayWeaponSound("Weapons/ss/tommygun/fire") //10
	    TNT1 A 0 A_FireBullets(0,0,1,(10),"PPE_SSBulletPuff",FBF_USEAMMO|FBF_NORANDOM|FBF_EXPLICITANGLE)
		SSTF ABC 1 bright
		TNT1 A 0 A_Refire
		Goto Ready
	Hold:
		SSTG A 1
		TNT1 A 0 A_GunFlash
		TNT1 A 0 A_AlertMonsters
		TNT1 A 0 A_PlayWeaponSound("Weapons/ss/tommygun/fire")
	    TNT1 A 0 A_FireBullets(0,0,1,(10),"PPE_SSBulletPuff",FBF_USEAMMO|FBF_NORANDOM|FBF_EXPLICITANGLE)
		SSTF DEF 1 bright
		TNT1 A 0 A_ClearRefire
		Goto Ready
	Flash:
		TNT1 A 1 A_Light1
		//TNT1 AA 1 A_SetPitch(pitch-0.4)
		TNT1 A 1 A_Light2
		//TNT1 A 1 A_SetPitch(pitch+0.8)
		Goto LightDone
	}
}
