Actor PPE_Q2Bfg : BFG9000
{
+WEAPON.NOAUTOAIM
+WEAPON.NOALERT
Inventory.Pickupmessage "$GOT_PPE_Q2BFG"
Tag "$TAG_PPE_Q2BFG"
Weapon.SelectionOrder 2800
Weapon.SlotNumber 0
Weapon.AmmoUse 5
Weapon.AmmoType "PPE_Nukes"
Inventory.PickupSound "items/weapon2"
Weapon.UpSound "weapons/switch"
Weapon.AmmoGive 5
Weapon.BobRangeX 0.05
Weapon.BobRangeY 0.75
Weapon.BobSpeed 2.0
States
	{
	Ready:
    	QBFG AABBCCDDEEFF 1 A_WeaponReady
    	Loop
  	Deselect:
    	QBFG A 1 A_Lower(18)
    	Loop
  	Select:
    	QBFG A 1 A_Raise(18)
    	Loop
	Fire:
		TNT1 A 0 A_JumpIfInventory("PowerSilencer",0,2)
		TNT1 A 0 A_AlertMonsters
		TNT1 A 0 A_PlaySound("items/quad/use",2,CountInv("PowerQuadDamage"))
		TNT1 A 0 A_PlaySound("weapons/bfg/fire2",1,1-(CountInv("PowerSilencer")*0.7))
 		QBFC AAABBBCCDDE 2 bright
		TNT1 A 0 A_GunFlash
		TNT1 A 0 A_PlaySound("items/quad/use",2,CountInv("PowerQuadDamage"))
		TNT1 A 0 A_FireCustomMissile("PPE_Q2BFGBall",0,1,0,4,1,0)
    	QBFF ABCD 3 bright
		QBFC A 3 bright
		QBFG AAAAAA 2
    	QBFR ABCDCBA 2
		QBFG AABBCCDDEEFF 1
		TNT1 A 0 A_ReFire
		Goto Ready
  	Flash:
    	TNT1 A 1 A_Light1
		TNT1 AAA 1 A_SetPitch(pitch-0.7)
    	TNT1 A 0 A_Light2
		TNT1 AAA 1 A_SetPitch(pitch-0.7)
		TNT1 A 0 A_Light(3)
		TNT1 AAA 1 A_SetPitch(pitch+0.7)
		TNT1 A 0 A_Light2
		TNT1 AAA 1 A_SetPitch(pitch+0.7)
		TNT1 A 0 A_Light1
    	Goto LightDone
  	Spawn:
		QBFW AAAAAABBCCDDCCBB 1
    	Loop
	}
}
