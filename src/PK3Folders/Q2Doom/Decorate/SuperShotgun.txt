Actor PPE_Q2SuperShotgun : SuperShotgun
{
	+WEAPON.NOALERT
	+WEAPON.NOAUTOAIM
	Inventory.Pickupmessage "$GOT_PPE_SSG"
	Tag "$TAG_PPE_SSG"
	Inventory.PickupSound "items/q2weapon"
	Weapon.SelectionOrder 400
	Weapon.SlotNumber 3
	Weapon.AmmoUse 2
	Weapon.AmmoType "PPE_Shells"
	Weapon.AmmoGive 8
	Decal "BulletChip"
	DamageType "Quake2Shell"
	States
	{
	Ready:
    	Q2SS A 1 A_WeaponReady
    	Loop
  	Deselect:
		Q2SS A 1 A_Lower
    	Loop
  	Select:
		Q2SS A 1 A_Raise
    	Loop
  	Fire:
		TNT1 A 0 A_GunFlash
		TNT1 A 0 A_JumpIfInventory("PowerSilencer",0,2)
		TNT1 A 0 A_AlertMonsters // 6 - should two shot cacodemons
		TNT1 A 0 A_FireBullets(7.3,3.9,20,(10),"PPE_BulletPuffQ2",FBF_USEAMMO|FBF_NORANDOM)
		TNT1 A 0 A_PlaySound("items/quad/amp",2,CountInv("PowerQuadDamage"))
		TNT1 A 0 A_PlaySound("weapons/supershotgun/fire",1,1-(CountInv("PowerSilencer")*0.7))
		Q2SS BCD 2 bright
		Q2SS A 4
		Q2SS EEEFFFGG 1
		Q2SS HHIIJJKKKJJII 1
		Q2SS HHGGFFEEA 1
		Goto Ready
  	Flash:
    	TNT1 A 1 A_Light1
		TNT1 AAA 1 A_SetPitch(pitch-0.61)
		TNT1 A 0 A_Light2
		TNT1 AAA 1 A_SetPitch(pitch-0.61)
		TNT1 A 0 A_Light1
		TNT1 AAA 1 A_SetPitch(pitch+1.22)
		Goto LightDone
  	Spawn:
    	Q2SS W -1
    	Stop
	}
}
