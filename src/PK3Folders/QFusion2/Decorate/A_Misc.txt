Actor PPE_GreenArmorSpawner : CustomInventory replaces GreenArmor
{
	States
	{
	Spawn:
		TNT1 A 0
		TNT1 A 0 A_JumpIf(CallAcs("GetLevelNum") >= 4,"Spawn2")
		TNT1 A 0 A_SpawnItem("PPE_GreenArmor")
		stop
	Spawn2:
		TNT1 A 0 A_Jump(256,1,2)
		TNT1 A 0 A_SpawnItem("PPE_GreenArmor")
		stop
		TNT1 A 0 A_SpawnItem("PPE_YellowArmor")
		stop
	}
}
