Actor DSS_Knife : Fist
{
	+WEAPON.NOALERT
	Inventory.Pickupmessage "$GOT_DSS_KN"
	Tag "$TAG_DSS_KN"
	Weapon.SlotNumber 1
	Weapon.Kickback 0
	DamageType "SeriousKnife"
	States
	{
	Ready:
    	SSKN A 1 A_WeaponReady
    	Loop
  	Deselect:
		SSKN A 1 A_Lower
		TNT1 A 0 A_Lower
    	Loop
  	Select:
		SSKN A 1 A_Raise
		TNT1 A 0 A_Raise
    	Loop
  	Fire:
		TNT1 A 0 A_AlertMonsters
		TNT1 A 0 A_GunFlash
		TNT1 A 0 A_PlayWeaponSound("weapons/knifeswing")
    	SSKN BC 2 // 120
		TNT1 A 0 A_FireBullets(0,0,1,(120),"DSS_KnifePuff",FBF_NORANDOM,72)
		SSKN DE 2
		SSKN F 21
		SSKN E 2
		TNT1 A 0 A_ReFire
		Goto Ready
  	Flash:
    	TNT1 A 1 A_Light1
		TNT1 AAAA 1 //A_SetPitch(pitch-0.4)
		TNT1 A 0 A_Light2
		TNT1 AA 1 //A_SetPitch(pitch+0.8)
		Goto LightDone
  	Spawn:
    	SSKN W -1
    	Stop
	}
}

Actor DSS_KnifePuff : BulletPuff
{
	+NOEXTREMEDEATH
	SeeSound "weapons/axeflesh"
	AttackSound "weapons/axehit"
	States
	{
	Spawn:
	Crash:
	Melee:
		TNT1 A 0
		PUFF AB 2 Bright
		PUFF CD 2
		Stop
  }
}
