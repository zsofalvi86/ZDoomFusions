Actor DSS_LaserGunBullet
{
	Radius 11
	Height 8
	Speed 20
	Damage 20
	Projectile
	Damage (24) // 20
	Speed 35
	SeeSound ""
	DeathSound "weapons/laserx"
	DamageType "SeriousCell"
	+BLOODSPLATTER
	States
	{
	Spawn:
		GLAP A 1 bright
		Loop
	Crash:
	Death:
		GPSX ABCDE 3 bright
		stop
	XDeath:
		TNT1 A 2
		stop
	}
}

Actor DSS_LaserGun : PlasmaRifle
{
	+WEAPON.NOALERT
	Inventory.Pickupmessage "$GOT_DSS_LG"
	Tag "$TAG_DSS_LG"
	Inventory.PickupSound "items/ssweapon"
	Weapon.SelectionOrder 110
	Weapon.SlotNumber 9
	Weapon.AmmoUse 1
	Weapon.AmmoType "DSS_Cells"
	Weapon.AmmoGive 40
	States
	{
	Spawn:
		SSLW A -1
		Stop
	Ready:
		SSLG A 1 A_WeaponReady
		Loop
	Deselect:
		SSLG A 1 A_Lower
		TNT1 A 0 A_Lower
		Loop
	Select:
		SSLG A 1 A_Raise
		TNT1 A 0 A_Raise
		Loop
	Fire:
	Fire1:
		TNT1 A 0 A_GunFlash
		TNT1 A 0 A_AlertMonsters
		TNT1 A 0 A_PlaySound("Weapons/laserf",1)
		TNT1 A 0 A_FireCustomMissile("DSS_LaserGunBullet",0,1,-5,4)
	    SSLF ABC 1 bright
		TNT1 A 0 A_Refire("Fire2")
		Goto Ready
	Fire2:
		TNT1 A 0 A_GunFlash
		TNT1 A 0 A_AlertMonsters
		TNT1 A 0 A_PlaySound("Weapons/laserf",2)
		TNT1 A 0 A_FireCustomMissile("DSS_LaserGunBullet",0,1,5,4)
		SSLF DEF 1 bright
		SSLG A 1
		TNT1 A 0 A_Refire("Fire3")
		goto Ready
	Fire3:
		TNT1 A 0 A_GunFlash
		TNT1 A 0 A_AlertMonsters
		TNT1 A 0 A_PlaySound("Weapons/laserf",3)
		TNT1 A 0 A_FireCustomMissile("DSS_LaserGunBullet",0,1,-5,-4)
		SSLF GHI 1 bright
		TNT1 A 0 A_Refire("Fire4")
		goto Ready
	Fire4:
		TNT1 A 0 A_GunFlash
		TNT1 A 0 A_AlertMonsters
		TNT1 A 0 A_PlaySound("Weapons/laserf",4)
		TNT1 A 0 A_FireCustomMissile("DSS_LaserGunBullet",0,1,5,-4)
		SSLF JKL 1 bright
		SSLG A 1
		TNT1 A 0 A_Refire("Fire1")
		goto Ready
	Flash:
		TNT1 A 0 A_Light1
		TNT1 AA 1 //A_SetPitch(pitch-0.4)
		TNT1 A 0 A_Light2
		TNT1 A 1 //A_SetPitch(pitch+0.8)
		Goto LightDone
	}
}
