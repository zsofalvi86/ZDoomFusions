{
	Species "BlueCard"
	Inventory.PickupMessage "$GOTBLUECARD"
	Inventory.Icon "STKEYS0"
	Inventory.PickupSound "items/keys/base"
	States
	{
	Spawn:
		BKEY A 10
		BKEY B 10 Bright
		Loop
